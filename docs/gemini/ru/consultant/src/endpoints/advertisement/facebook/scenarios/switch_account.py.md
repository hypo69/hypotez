"""
Модуль для переключения между аккаунтами в Facebook.
=======================================================

Модуль содержит функцию :func:`switch_account`, которая выполняет переключение между аккаунтами,
если соответствующая кнопка доступна.

Пример использования
----------------------

>>> from src.webdriver.driver import Driver
>>> # Предполагается, что у вас уже есть инстанс драйвера
>>> # driver = Driver(Chrome)
>>> switch_account(driver)
"""

from pathlib import Path
from types import SimpleNamespace
from src import gs
from src.webdriver.driver import Driver
from src.utils.jjson import j_loads_ns
from src.logger import logger

# Загрузка локаторов из JSON файла.
locator: SimpleNamespace = j_loads_ns(
    Path(gs.path.src / 'endpoints' / 'advertisement' / 'facebook' / 'locators' / 'post_message.json')
)

def switch_account(driver: Driver) -> None:
    """
    Выполняет переключение на другой аккаунт, если доступна кнопка "Переключить".

    Args:
        driver (Driver): Инстанс веб-драйвера.

    Returns:
        None

    Raises:
        Exception: Если происходит ошибка при выполнении локатора.

    Example:
        >>> from src.webdriver.driver import Driver
        >>> # Предполагается, что у вас уже есть инстанс драйвера
        >>> # driver = Driver(Chrome)
        >>> switch_account(driver)
    """
    try:
        # Пытаемся выполнить локатор для кнопки переключения аккаунта
        driver.execute_locator(locator.switch_to_account_button)
        logger.info('Успешно выполнено переключение аккаунта.')
    except Exception as ex:
        # Логируем ошибку, если что-то пошло не так
        logger.error('Ошибка при переключении аккаунта', ex, exc_info=True)
