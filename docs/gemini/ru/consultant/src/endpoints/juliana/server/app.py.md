import os
from flask import Flask

"""
Модуль инициализации Flask-приложения для сервиса Juliana.
===========================================================

Этот модуль содержит код для создания и настройки Flask-приложения,
предназначенного для обработки запросов к сервису Juliana.

Пример использования
----------------------
>>> app = create_app()
>>> app.run(debug=True)
"""


def create_app() -> Flask:
    """
    Создает и настраивает Flask-приложение.

    Args:
        None

    Returns:
        Flask: Объект Flask-приложения.
    """
    app = Flask(__name__, template_folder=os.path.abspath('./../client/html'))
    return app


app = create_app()
