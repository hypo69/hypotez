"""
Основной модуль для запуска графического интерфейса G4F (GPT4Free) на платформе Windows.
====================================================================================

Этот модуль инициализирует и запускает графический интерфейс G4F, используя библиотеку `g4f.gui`.
Он также настраивает SSL для работы с сертификатами и обрабатывает аргументы командной строки.

Пример использования:
----------------------
>>> python main.py --theme dark --port 8000
"""

import ssl
import certifi
from functools import partial
from argparse import ArgumentParser

from src.logger import logger  # Импорт модуля логгера
from g4f.gui.run import run_gui_args, gui_parser
import g4f.debug

ssl.default_ca_certs = certifi.where()
ssl.create_default_context = partial(
    ssl.create_default_context,
    cafile=certifi.where()
)

g4f.debug.version_check = False
g4f.debug.version = "0.3.1.7"


def main():
    """
    Основная функция для запуска графического интерфейса G4F.

    Инициализирует парсер аргументов, обрабатывает аргументы командной строки
    и запускает графический интерфейс.
    """
    try:
        parser: ArgumentParser = gui_parser()  # Создание парсера аргументов
        args = parser.parse_args()  # Обработка аргументов командной строки
        run_gui_args(args)  # Запуск графического интерфейса с переданными аргументами
    except Exception as ex:
        logger.error('Произошла ошибка при запуске GUI', ex, exc_info=True)


if __name__ == "__main__":
    main()
