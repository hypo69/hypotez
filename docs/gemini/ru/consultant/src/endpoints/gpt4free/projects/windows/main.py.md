"""
Модуль для запуска графического интерфейса g4f.
=================================================

Модуль является точкой входа для запуска графического интерфейса (GUI) g4f (gpt4free). 
Он настраивает SSL, отключает проверку версий и запускает GUI.

Пример использования
----------------------

>>> python main.py --port 8000
"""

import ssl
import certifi
from functools import partial
from g4f.gui.run import run_gui_args, gui_parser
import g4f.debug
from argparse import Namespace
from src.logger import logger

# Настройка SSL для использования certifi
ssl.default_ca_certs = certifi.where()
ssl.create_default_context = partial(
    ssl.create_default_context,
    cafile=certifi.where()
)

# Отключение проверки версий и установка версии для отладки
g4f.debug.version_check = False
g4f.debug.version = "0.3.1.7"


def main():
    """
    Главная функция для запуска GUI g4f.
    Настраивает парсер аргументов, парсит аргументы и запускает GUI.
    """
    try:
        parser = gui_parser()
        args: Namespace = parser.parse_args()
        run_gui_args(args)
    except Exception as ex:
        logger.error('Error while running GUI', ex, exc_info=True)


if __name__ == "__main__":
    main()
