**Оптимизированный код**:

```python
"""
Модуль для генерации файла openapi.json
=========================================

Модуль использует библиотеку g4f для создания приложения и получения схемы OpenAPI,
затем сохраняет эту схему в файл openapi.json.
"""
import json
from typing import Dict
from src.logger import logger

from g4f.api import create_app

app = create_app()

try:
    data: Dict = app.openapi() # Получаем описание OpenAPI схемы
    json_data: str = json.dumps(data) # Конвертируем схему в JSON-формат

    with open('openapi.json', 'w', encoding='utf-8') as f:
        f.write(json_data) # Записываем JSON-данные в файл

    file_size_kb: float = round(len(json_data) / 1024, 2)
    logger.info(f'openapi.json - {file_size_kb} kbytes') # Логируем размер файла

except Exception as ex:
    logger.error('Ошибка при создании или записи файла openapi.json', ex, exc_info=True)
