```python
import asyncio
from g4f.client import AsyncClient
from src.logger import logger  # Импорт модуля logger


async def main():
    """
    Асинхронная функция для демонстрации работы с GPT-4o через библиотеку g4f.

    Отправляет запрос к модели и выводит полученный ответ.
    """
    try:
        client: AsyncClient = AsyncClient()  # Создание асинхронного клиента
        logger.info('AsyncClient создан') # Логирование информации об успешном создании клиента

        response = await client.chat.completions.create(  # Отправка запроса к API
            model="gpt-4o",
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": "how does a court case get to the Supreme Court?"}
            ]
        )

        if response and response.choices:  # Проверка наличия ответа и choices
            print(response.choices[0].message.content)  # Вывод ответа модели
            logger.info('Ответ от модели получен и выведен')  # Логирование информации об успешном получении и выводе ответа
        else:
            logger.warning('Пустой ответ от API')  # Логирование предупреждения о пустом ответе

    except Exception as ex:
        logger.error('Произошла ошибка при выполнении запроса к API', ex, exc_info=True)  # Логирование ошибки
        print(f"Ошибка: {ex}")  # Вывод сообщения об ошибке


if __name__ == "__main__":
    asyncio.run(main())  # Запуск асинхронной функции
