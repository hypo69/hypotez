"""
Модуль для демонстрации асинхронного получения текстовых завершений с использованием g4f.
==========================================================================

Модуль содержит асинхронную функцию `main`, которая демонстрирует использование AsyncClient
из библиотеки g4f для получения ответа от модели gpt-4o на заданный вопрос.

Пример использования
----------------------

>>> asyncio.run(main())
"""

import asyncio
from g4f.client import AsyncClient
from src.logger import logger


async def main() -> None:
    """
    Асинхронно получает текстовое завершение от модели gpt-4o.

    Args:
        None

    Returns:
        None

    Raises:
        Exception: Если происходит ошибка при взаимодействии с API g4f.

    """
    try:
        client: AsyncClient = AsyncClient()  # Создание асинхронного клиента
        response = await client.chat.completions.create(  # Отправка запроса на получение ответа
            model="gpt-4o",
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": "how does a court case get to the Supreme Court?"}
            ]
        )

        logger.info(response.choices[0].message.content)  # Вывод полученного ответа
    except Exception as ex:
        logger.error('Error while processing data', ex, exc_info=True)


asyncio.run(main())  # Запуск асинхронной функции main
