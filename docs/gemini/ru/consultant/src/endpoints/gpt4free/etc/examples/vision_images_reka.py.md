"""
Модуль для работы с image chat Reka
====================================

Модуль содержит пример использования API Reka для анализа изображений.
Требуется наличие cookies и авторизация на chat.reka.ai.

Пример использования
----------------------

>>> # Сначала необходимо установить библиотеку g4f
>>> # pip install g4f
>>> # Затем поместите изображение 'cat.jpeg' в папку 'docs/images/'
>>> # Запустите скрипт vision_images_reka.py
"""

from g4f.client import Client
from g4f.Provider import Reka
from src.logger import logger  # Подключаем модуль логирования
from pathlib import Path


def analyze_image(image_path: str | Path) -> None:
    """
    Анализирует изображение с использованием API Reka и выводит результат.

    Args:
        image_path (str | Path): Путь к файлу изображения.

    Returns:
        None

    Raises:
        FileNotFoundError: Если файл изображения не найден.
        Exception: Если возникает ошибка при взаимодействии с API Reka.

    Example:
        >>> analyze_image("docs/images/cat.jpeg")
        Результат анализа изображения: ...
    """
    try:
        # Проверяем наличие файла изображения
        image_path = Path(image_path).resolve()
        if not image_path.exists():
            raise FileNotFoundError(f'Файл изображения не найден: {image_path}')

        # Создание клиента для взаимодействия с API Reka
        client = Client(provider=Reka)

        # Отправка запроса на анализ изображения
        completion = client.chat.completions.create(
            model='reka-core',
            messages=[
                {
                    'role': 'user',
                    'content': 'What can you see in the image ?'
                }
            ],
            stream=True,
            image=open(str(image_path), 'rb')  # Открываем файл изображения в бинарном режиме
        )

        # Вывод результата анализа
        print('Результат анализа изображения:')
        for message in completion:
            print(message.choices[0].delta.content or '', end='')
        print()  # Добавляем перенос строки в конце вывода

    except FileNotFoundError as ex:
        logger.error(f'Файл не найден: {image_path}', ex, exc_info=True)
        print(f'Ошибка: Файл не найден: {image_path}')
    except Exception as ex:
        logger.error('Ошибка при анализе изображения', ex, exc_info=True)
        print(f'Произошла ошибка: {ex}')


if __name__ == '__main__':
    image_file = 'docs/images/cat.jpeg'
    analyze_image(image_file)
