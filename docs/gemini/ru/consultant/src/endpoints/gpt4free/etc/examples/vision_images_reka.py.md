"""
Модуль для демонстрации анализа изображений с использованием модели Reka.
==========================================================================

Этот модуль показывает, как использовать клиент G4F для взаимодействия с моделью Reka
и отправлять запросы на анализ изображений.

Пример использования:
--------------------

>>> # Перед использованием необходимо установить библиотеку g4f
>>> # pip install g4f
>>> # Убедитесь, что у вас есть файл изображения 'docs/images/cat.jpeg'.
>>> # Запустите скрипт, чтобы получить описание изображения от модели Reka.
"""

from g4f.client import Client
from g4f.Provider import Reka
from src.logger import logger  # Подключаем модуль логгирования
from typing import Generator

def analyze_image(image_path: str) -> Generator[str, None, None] | None:
    """
    Анализирует изображение с использованием модели Reka и возвращает ответ.

    Args:
        image_path (str): Путь к файлу изображения.

    Returns:
        Generator[str, None, None] | None: Ответ от модели Reka в виде генератора или None в случае ошибки.

    Raises:
        FileNotFoundError: Если файл изображения не найден.
        Exception: Если происходит ошибка при взаимодействии с API Reka.

    Example:
        >>> image_path = "docs/images/cat.jpeg"
        >>> response = analyze_image(image_path)
        >>> if response:
        ...     for chunk in response:
        ...         print(chunk, end="")
        ...     print()
        ... else:
        ...     print("Не удалось получить ответ от модели.")
    """
    try:
        # Создание клиента для взаимодействия с моделью Reka
        client: Client = Client(provider=Reka)  # Optional if you set model name to reka-core

        # Открытие файла изображения в бинарном режиме
        with open(image_path, "rb") as image_file:
            # Отправка запроса на анализ изображения
            completion: Generator = client.chat.completions.create(
                model="reka-core",
                messages=[
                    {
                        "role": "user",
                        "content": "What can you see in the image ?"  # Запрос к модели
                    }
                ],
                stream=True,
                image=image_file  # Передача файла изображения
            )

            return completion

    except FileNotFoundError as ex:
        logger.error(f"Файл изображения не найден: {image_path}", ex, exc_info=True)  # Логирование ошибки
        return None
    except Exception as ex:
        logger.error("Ошибка при взаимодействии с API Reka", ex, exc_info=True)  # Логирование ошибки
        return None

if __name__ == "__main__":
    image_path: str = "docs/images/cat.jpeg"  # Путь к файлу изображения
    response: Generator[str, None, None] | None = analyze_image(image_path)  # Получение ответа от модели

    if response:
        for chunk in response:
            print(chunk, end="")  # Вывод ответа чанками
        print()
    else:
        print("Не удалось получить ответ от модели.")  # Вывод сообщения об ошибке
