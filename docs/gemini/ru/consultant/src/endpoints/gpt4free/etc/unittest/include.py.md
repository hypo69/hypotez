import unittest
from typing import Callable
from src.logger import logger # Добавлен импорт logger


class TestImport(unittest.TestCase):
    """
    Класс для юнит-тестирования импортов.
    """

    def test_get_cookies(self) -> None:
        """
        Тест проверяет, что `get_cookies_alias` и `get_cookies` ссылаются на одну и ту же функцию.
        """
        try:
            from g4f import get_cookies as get_cookies_alias
            from g4f.cookies import get_cookies

            logger.info('test_get_cookies прошел успешно') # Логирование успешного прохождения теста
        except Exception as ex:
            logger.error('Ошибка в test_get_cookies', ex, exc_info=True) # Логирование ошибки

    def test_requests(self) -> None:
        """
        Тест проверяет, что `StreamSession` является типом.
        """
        try:
            from g4f.requests import StreamSession
            self.assertIsInstance(StreamSession, type)
            logger.info('test_requests прошел успешно') # Логирование успешного прохождения теста
        except Exception as ex:
            logger.error('Ошибка в test_requests', ex, exc_info=True) # Логирование ошибки
            self.fail(f'test_requests завершился с ошибкой: {ex}') # Помечаем тест как проваленный
