"""
Модуль для запуска графического интерфейса G4F
================================================

Этот модуль содержит код для запуска графического интерфейса G4F, 
используя функцию run_gui из модуля g4f.gui.

Пример использования
----------------------

>>> from g4f.gui import run_gui
>>> run_gui()
"""

import sys
from pathlib import Path

from src.logger import logger  # Подключаем модуль логгирования

# Добавляем в sys.path директорию, содержащую модуль g4f, для корректного импорта
try:
    sys.path.append(str(Path(__file__).parent.parent.parent))

    from g4f.gui import run_gui


    def start_gui() -> None:
        """
        Запускает графический интерфейс G4F.
        """
        try:
            run_gui()  # Запускаем графический интерфейс
        except Exception as ex:
            logger.error('Ошибка при запуске графического интерфейса', ex, exc_info=True)  # Логируем ошибку, если что-то пошло не так


    if __name__ == "__main__":
        start_gui()
except Exception as ex:
    logger.error('Ошибка при добавлении пути или импорте модуля g4f.gui', ex, exc_info=True)
