"""
Модуль для обработки логирования и перехвата исключений
========================================================

Модуль содержит функции для настройки обработки исключений и интеграции с системой логирования.
"""

import sys
import logging
from typing import Any, Type
from src.logger import logger  # Подключаем модуль логирования

def __exception_handle(e_type: Type[BaseException], e_value: BaseException, e_traceback: Any) -> None:
    """
    Обработчик необработанных исключений.

    Args:
        e_type (Type[BaseException]): Тип исключения.
        e_value (BaseException): Значение исключения.
        e_traceback (Any): Stack trace.
    """
    if issubclass(e_type, KeyboardInterrupt):
        print('\nBye...')
        sys.exit(0)

    logger.error('Необработанное исключение', exc_info=(e_type, e_value, e_traceback))  # Логируем исключение
    sys.__excepthook__(e_type, e_value, e_traceback)

def hook_except_handle() -> None:
    """
    Устанавливает обработчик исключений.
    """
    sys.excepthook = __exception_handle
