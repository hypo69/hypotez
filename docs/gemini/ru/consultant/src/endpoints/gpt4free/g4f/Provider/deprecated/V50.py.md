from __future__ import annotations

import uuid
import requests
from typing import Any, CreateResult, List, Dict
from src.logger import logger # Импорт модуля logger
from ..base_provider import AbstractProvider


class V50(AbstractProvider):
    """
    Провайдер V50 для взаимодействия с API p5.v50.ltd.
    Поддерживает модель gpt-3.5-turbo.
    УСТАРЕЛ: Этот модуль больше не поддерживается и может быть удален в будущем.
    """
    url: str = 'https://p5.v50.ltd'
    supports_gpt_35_turbo: bool = True
    supports_stream: bool = False
    needs_auth: bool = False
    working: bool = False

    @staticmethod
    def create_completion(
        model: str,
        messages: List[Dict[str, str]],
        stream: bool, 
        **kwargs: Any
    ) -> CreateResult:
        """
        Создает запрос к API p5.v50.ltd для получения ответа от языковой модели.

        Args:
            model (str): Имя модели.
            messages (List[Dict[str, str]]): Список сообщений для отправки в API.
            stream (bool): Флаг стриминга (не поддерживается).
            **kwargs (Any): Дополнительные параметры (температура, top_p, proxy).

        Returns:
            CreateResult: Результат запроса.
        
        Raises:
            requests.exceptions.RequestException: При ошибке запроса к API.
            Exception: При возникновении других ошибок.
        """
        try:
            conversation: str = (
                "\n".join(
                    f"{message['role']}: {message['content']}" for message in messages
                )
                + "\nassistant: "
            )
            payload: Dict[str, Any] = {
                "prompt": conversation,
                "options": {},
                "systemMessage": ".",
                "temperature": kwargs.get("temperature", 0.4),
                "top_p": kwargs.get("top_p", 0.4),
                "model": model,
                "user": str(uuid.uuid4())
            }

            headers: Dict[str, str] = {
                'authority': 'p5.v50.ltd',
                'accept': 'application/json, text/plain, */*',
                'accept-language': 'id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7',
                'content-type': 'application/json',
                'origin': 'https://p5.v50.ltd',
                'referer': 'https://p5.v50.ltd/',
                'sec-ch-ua-platform': '"Windows"',
                'sec-fetch-dest': 'empty',
                'sec-fetch-mode': 'cors',
                'sec-fetch-site': 'same-origin',
                'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/116.0.0.0 Safari/537.36'
            }
            logger.info(f"Отправка запроса к {V50.url}") # Логирование отправки запроса
            response = requests.post(
                "https://p5.v50.ltd/api/chat-process",
                json=payload,
                headers=headers,
                proxies=kwargs.get('proxy', {}),
            )
            logger.info(f"Получен ответ от {V50.url}") # Логирование получения ответа

            if "https://fk1.v50.ltd" not in response.text:
                yield response.text
        except requests.exceptions.RequestException as ex:
            logger.error(f"Ошибка при запросе к API {V50.url}", ex, exc_info=True) # Логирование ошибки запроса
            raise
        except Exception as ex:
            logger.error("Произошла ошибка", ex, exc_info=True) # Логирование общей ошибки
            raise
