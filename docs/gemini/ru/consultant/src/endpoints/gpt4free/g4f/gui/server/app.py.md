```python
import sys
import os
from flask import Flask
from src.logger import logger  # Импорт модуля logger


def create_app() -> Flask:
    """
    Создает и конфигурирует Flask-приложение.

    Функция определяет местоположение шаблонов в зависимости от того,
    запущено ли приложение в "замороженном" состоянии (например, как исполняемый файл)
    или в режиме разработки, и создает экземпляр Flask-приложения с соответствующими настройками.

    Returns:
        Flask: Сконфигурированное Flask-приложение.
    """
    try:
        if getattr(sys, 'frozen', False):
            # Если приложение запущено как исполняемый файл
            template_folder = os.path.join(sys._MEIPASS, 'client')
        else:
            # Если приложение запущено в режиме разработки
            template_folder = '../client'

        # Создание Flask-приложения с указанием папки для шаблонов и статических файлов
        app = Flask(__name__, template_folder=template_folder, static_folder=f'{template_folder}/static')
        # Включение автоматической перезагрузки шаблонов в режиме отладки
        app.config['TEMPLATES_AUTO_RELOAD'] = True
        logger.info('Flask app created successfully')  # Логирование успешного создания приложения
        return app
    except Exception as ex:
        logger.error('Error while creating Flask app', ex, exc_info=True)  # Логирование ошибки при создании приложения
        raise
