import sys
import os
from flask import Flask
from src.logger import logger


def create_app() -> Flask:
    """
    Создает и конфигурирует Flask-приложение для gpt4free GUI.

    Функция определяет путь к шаблонам в зависимости от того, запущено ли приложение в замороженном состоянии
    (например, как исполняемый файл) или в режиме разработки.
    Автоматически включает перезагрузку шаблонов в режиме отладки.

    Returns:
        Flask: Сконфигурированное Flask-приложение.

    Raises:
        OSError: Если не удается определить путь к шаблонам в замороженном состоянии.
    """
    try:
        if getattr(sys, 'frozen', False):
            # Если приложение запущено как исполняемый файл, ищем шаблоны в директории MEIPASS
            template_folder = os.path.join(sys._MEIPASS, 'client')
            logger.info(f'Running in frozen mode. Template folder: {template_folder}')  # Логируем информацию
        else:
            # Иначе, запускаем в режиме разработки, шаблоны находятся в относительной директории
            template_folder = '../client'
            logger.info(f'Running in development mode. Template folder: {template_folder}')  # Логируем информацию

        app = Flask(__name__, template_folder=template_folder, static_folder=f'{template_folder}/static')
        app.config['TEMPLATES_AUTO_RELOAD'] = True  # Включаем авто-перезагрузку шаблонов
        logger.info('Templates auto reload is enabled')  # Логируем информацию

        return app

    except OSError as ex:
        logger.error('Error while creating the app', ex, exc_info=True)  # Логируем ошибку
        raise  # Перебрасываем исключение дальше
