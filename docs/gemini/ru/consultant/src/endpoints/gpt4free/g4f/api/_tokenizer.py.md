"""
Модуль для токенизации текста с использованием библиотеки tiktoken
==================================================================

Модуль предоставляет функцию `tokenize`, которая использует библиотеку `tiktoken` для кодирования текста
и подсчета количества токенов.

Пример использования:
----------------------

>>> num_tokens, encoded = tokenize("Пример текста для токенизации.", model='gpt-3.5-turbo')
>>> print(f"Количество токенов: {num_tokens}")
>>> print(f"Закодированный текст: {encoded}")
"""

import tiktoken
from typing import List
from src.logger import logger


def tokenize(text: str, model: str = 'gpt-3.5-turbo') -> tuple[int, List[int]] | None:
    """
    Выполняет токенизацию текста с использованием указанной модели.

    Args:
        text (str): Текст для токенизации.
        model (str, optional): Модель tiktoken для использования. По умолчанию 'gpt-3.5-turbo'.

    Returns:
        tuple[int, List[int]] | None: Кортеж, содержащий количество токенов и список закодированных токенов,
        или None в случае ошибки.

    Raises:
        Exception: Если происходит ошибка при кодировании текста.

    Example:
        >>> text = "Пример текста для токенизации."
        >>> num_tokens, encoded = tokenize(text, model='gpt-3.5-turbo')
        >>> if num_tokens is not None and encoded is not None:
        ...     print(f"Количество токенов: {num_tokens}")
        ...     print(f"Закодированный текст: {encoded}")
        Количество токенов: 6
        Закодированный текст: [94282, 11136, 1794, 6403, 73148, 13]
    """
    try:
        encoding = tiktoken.encoding_for_model(model) # Получение кодировщика для указанной модели
        encoded = encoding.encode(text) # Кодирование текста
        num_tokens = len(encoded) # Вычисление количества токенов
        return num_tokens, encoded # Возврат количества токенов и закодированного текста
    except Exception as ex:
        logger.error('Ошибка при токенизации текста', ex, exc_info=True) # Логирование ошибки
        return None
