"""
Модуль для токенизации текста
==============================

Модуль содержит функции для токенизации текста с использованием библиотеки `tiktoken`.
"""
import tiktoken
from typing import Union
from src.logger import logger

def tokenize(text: str, model: str = 'gpt-3.5-turbo') -> Union[int, str]:
    """
    Токенизирует текст с использованием указанной модели.

    Args:
        text (str): Текст для токенизации.
        model (str, optional): Модель для токенизации. Defaults to 'gpt-3.5-turbo'.

    Returns:
        Union[int, str]: Количество токенов в тексте и закодированный текст.
        Возвращает `None` в случае ошибки.

    Raises:
        Exception: Если возникает ошибка при токенизации текста.

    Example:
        >>> tokenize("Hello world!", model='gpt-3.5-turbo')
        (3, [15496, 992, 0])
    """
    try:
        encoding = tiktoken.encoding_for_model(model)
        encoded = encoding.encode(text)
        num_tokens = len(encoded)

        return num_tokens, encoded
    except Exception as ex:
        logger.error('Error while tokenizing text', ex, exc_info=True)
        return None
