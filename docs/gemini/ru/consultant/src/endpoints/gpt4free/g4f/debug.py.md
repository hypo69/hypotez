"""
Модуль для отладки и логирования в g4f
=========================================

Модуль содержит функции для управления логированием, обработки ошибок и проверки версий.
"""
import sys
from typing import Callable, List, Optional, Any

from src.logger import logger  # Import logger

logging: bool = False
"""Флаг, определяющий, включено ли логирование."""
version_check: bool = True
"""Флаг, определяющий, следует ли выполнять проверку версии."""
version: Optional[str] = None
"""Текущая версия."""
log_handler: Callable = print
"""Обработчик для логирования. По умолчанию используется функция print."""
logs: List[str] = []
"""Список для хранения логов."""


def log(*text: Any, file: Optional[Any] = None) -> None:
    """
    Логирует сообщение, если логирование включено.

    Args:
        *text (Any): Сообщения для логирования.
        file (Optional[Any], optional): Файл для записи логов. По умолчанию `None`.
    """
    try:  # add exception handling
        if logging:
            log_handler(*text, file=file)
    except Exception as ex:
        logger.error(f'Ошибка при логировании: {ex}', exc_info=True)  # use logger.error


def error(*error: Any, name: Optional[str] = None) -> None:
    """
    Логирует сообщение об ошибке в stderr.

    Args:
        *error (Any): Сообщения об ошибке.
        name (Optional[str], optional): Имя ошибки. По умолчанию `None`.
    """
    try:  # add exception handling
        error = [e if isinstance(e, str) else f'{type(e).__name__ if name is None else name}: {e}' for e in error]
        log(*error, file=sys.stderr)
    except Exception as ex:
        logger.error(f'Ошибка при логировании ошибки: {ex}', exc_info=True)  # use logger.error
