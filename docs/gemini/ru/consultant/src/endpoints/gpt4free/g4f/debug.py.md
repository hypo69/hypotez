"""
Модуль для отладки и логирования g4f.
=======================================

Модуль содержит функции для логирования сообщений и ошибок, а также глобальные переменные
для управления логированием и проверки версий.
"""
import sys
from typing import Callable, List, Optional, Any

from src.logger import logger  # Импорт модуля логирования

logging: bool = False
version_check: bool = True
version: Optional[str] = None
log_handler: Callable = print  # More specifically: Callable[[Any, Optional[Any]], None]
logs: List[str] = []


def log(*text: Any, file: Optional[Any] = None) -> None:
    """
    Логирует сообщение, если логирование включено.

    Args:
        *text (Any): Переменное количество аргументов для логирования.
        file (Optional[Any], optional): Файл для записи логов. По умолчанию None (stdout).

    Example:
        >>> log('This is a log message')
    """
    if logging:
        log_handler(*text, file=file)


def error(*error: Any, name: Optional[str] = None) -> None:
    """
    Логирует сообщение об ошибке в stderr.

    Args:
        *error (Any): Переменное количество аргументов для сообщения об ошибке.
        name (Optional[str], optional): Имя ошибки. По умолчанию None.

    Example:
        >>> error('This is an error message', name='ValueError')
    """
    # Преобразуем каждый элемент ошибки в строку, если это не строка
    error = [
        e if isinstance(e, str) else f"{type(e).__name__ if name is None else name}: {e}"
        for e in error
    ]
    logger.error(*error, file=sys.stderr)  # Используем logger.error для логирования ошибок
