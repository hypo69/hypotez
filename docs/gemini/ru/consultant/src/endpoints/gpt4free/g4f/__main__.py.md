"""
Модуль является точкой входа для запуска GPT4Free API.
========================================================

Он использует аргументы командной строки для конфигурации и запуска API.

Пример использования
----------------------

>>> python -m g4f --help
"""
from __future__ import annotations

from src.logger import logger # Импорт модуля логирования
from .cli import get_api_parser, run_api_args
import argparse # Импорт argparse

def main() -> None:
    """
    Точка входа для запуска GPT4Free API.

    Функция парсит аргументы командной строки и запускает API.
    """
    parser = get_api_parser()
    try:
        args = parser.parse_args()
        if args.gui is None:
            args.gui = True
        run_api_args(args)
    except argparse.ArgumentError as ex:
        logger.error('Ошибка парсинга аргументов командной строки', ex, exc_info = True) # Логирование ошибки парсинга аргументов
    except Exception as ex:
        logger.error('Произошла ошибка при запуске API', ex, exc_info = True) # Логирование общей ошибки запуска API


if __name__ == "__main__":
    main()
