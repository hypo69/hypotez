**Оптимизированный код:**

```python
"""
Модуль для запуска g4f API.
==========================

Модуль содержит точку входа для запуска API g4f, обрабатывает аргументы командной строки и запускает API.

Пример использования:
----------------------
>>> python -m g4f --help
"""
from __future__ import annotations

from src.logger import logger # Импорт модуля logger для логирования
from .cli import get_api_parser, run_api_args

def main():
    """
    Основная функция для запуска API g4f.

    Настраивает и запускает API на основе аргументов командной строки.
    """
    parser = get_api_parser() # Получаем парсер аргументов командной строки
    args = parser.parse_args() # Парсим аргументы командной строки

    if args.gui is None: # Если аргумент gui не задан
        args.gui = True # Устанавливаем значение gui в True по умолчанию

    try:
        run_api_args(args) # Запускаем API с переданными аргументами
    except Exception as ex: # Ловим возможные исключения
        logger.error('Ошибка при запуске API', ex, exc_info=True) # Логируем ошибку

if __name__ == "__main__":
    main()
