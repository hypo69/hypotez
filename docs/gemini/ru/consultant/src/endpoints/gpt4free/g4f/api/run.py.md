"""
Модуль для запуска API g4f
=================================================

Модуль содержит функцию :func:`run_api`, которая используется для запуска API g4f
с возможностью отладки.

Пример использования
----------------------

>>> import g4f.api
>>> g4f.api.run_api(debug=True)
"""

import g4f.api
from src.logger import logger  # Import logger

def run_api(debug: bool = False) -> None:
    """
    Запускает API g4f с возможностью отладки.

    Args:
        debug (bool, optional): Флаг, указывающий, следует ли запускать API в режиме отладки. По умолчанию False.

    Returns:
        None

    Raises:
        Exception: Если при запуске API происходит ошибка.
    """
    try:
        g4f.api.run_api(debug=debug) # Запуск API g4f
    except Exception as ex:
        logger.error('Ошибка при запуске API g4f', ex, exc_info=True) # Логирование ошибки
        raise # Переброс исключения для дальнейшей обработки


if __name__ == "__main__":
    run_api(debug=True) # Запуск API в режиме отладки, если скрипт запущен напрямую
