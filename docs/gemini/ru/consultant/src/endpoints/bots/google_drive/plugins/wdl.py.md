import os
import subprocess
from typing import Tuple
from src.logger import logger


def wget_dl(url: str) -> Tuple[str, str]:
    """
    Скачивает файл по указанному URL, используя утилиту wget.

    Args:
        url (str): URL для скачивания файла.

    Returns:
        Tuple[str, str]: Кортеж, содержащий статус скачивания ("ok" или "error") и имя скачанного файла.
                         В случае ошибки возвращает ("error", filename).

    Raises:
        FileNotFoundError: Если утилита wget не найдена в системе.
        subprocess.CalledProcessError: Если команда wget завершается с ненулевым кодом возврата.
        Exception: В случае любой другой ошибки.

    Example:
        >>> wget_dl('https://example.com/file.txt')
        ('ok', 'file.txt')
    """
    filename = os.path.basename(url)  # Извлекаем имя файла из URL
    try:
        logger.info(f'Начало скачивания файла {filename} по URL: {url}')  # Логируем начало скачивания
        subprocess.check_output(
            ['wget', '--output-document', filename, url],
            stderr=subprocess.STDOUT,
            shell=False,
            executable='/bin/bash', # Добавил указание bash
            timeout=30 # Добавил timeout
        )  # Выполняем команду wget для скачивания файла
        logger.info(f'Скачивание файла {filename} завершено')  # Логируем успешное завершение скачивания
        return "ok", filename  # Возвращаем статус "ok" и имя файла
    except FileNotFoundError as ex:  # Обрабатываем исключение, если wget не найден
        logger.error('Утилита wget не найдена в системе', ex, exc_info=True)  # Логируем ошибку
        return "error", filename  # Возвращаем статус "error" и имя файла
    except subprocess.CalledProcessError as ex:  # Обрабатываем исключение, если команда завершилась с ошибкой
        logger.error(f'Ошибка при скачивании файла {filename}: {ex.output.decode()}', ex, exc_info=True)  # Логируем ошибку
        return "error", filename  # Возвращаем статус "error" и имя файла
    except Exception as ex:  # Обрабатываем все остальные исключения
        logger.error(f'Неизвестная ошибка при скачивании файла {filename}', ex, exc_info=True)  # Логируем ошибку
        return "error", filename  # Возвращаем статус "error" и имя файла
