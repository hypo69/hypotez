"""
Модуль для идентификации токенов
=======================================

Модуль содержит функцию :func:`is_token`, которая определяет, является ли входная строка токеном заданного формата.

Пример использования
----------------------

>>> is_token("some_string / some_other_string")
False
"""
from src.logger import logger


def is_token(token: str) -> bool:
    """
    Проверяет, является ли предоставленная строка токеном определенного формата.

    Args:
        token (str): Строка, которую необходимо проверить.

    Returns:
        bool: True, если строка является токеном, иначе False.

    Example:
        >>> is_token("A1 / some_other_string")
        False
    """
    try:
        token_part = token.split()[-1]  # Разделяем строку и берем последний элемент
        token_length = len(token_part)  # Определяем длину последнего элемента

        if token_length == 62:  # Проверяем, соответствует ли длина токена требуемой
            if token_part[1] == '/':  # Проверяем наличие символа '/' на второй позиции
                logger.info("Токен успешно идентифицирован")  # Логируем успешное обнаружение токена
                return True
            else:
                logger.warning("Токен не соответствует формату: отсутствует символ '/'")  # Логируем несоответствие формата
                return False
        else:
            logger.warning(f"Токен не соответствует формату: неправильная длина {token_length}")  # Логируем несоответствие длины
            return False
    except Exception as ex:
        logger.error('Ошибка при обработке токена', ex, exc_info=True)  # Логируем ошибку при обработке токена
        return False
