```python
"""
Модуль для проверки формата токена
=====================================

Модуль содержит функцию :func:`is_token`, которая проверяет, соответствует ли токен заданному формату.
"""
from src.logger import logger

def is_token(token: str) -> bool:
    """
    Проверяет, является ли строка валидным токеном.

    Args:
        token (str): Строка, которую необходимо проверить.

    Returns:
        bool: True, если строка является валидным токеном, иначе False.

    Example:
        >>> is_token('some_token_with_slash')
        False
    """
    # Проверяем, что токен не пустой
    if not token:
        logger.warning('Токен не может быть пустым')
        return False

    token = token.split()[-1]
    TLEN = len(token)

    # Валидация длинны токена
    if TLEN == 62:
        # Проверяем наличие символа "/" на второй позиции (индекс 1)
        if token[1] == '/':
            return True
        else:
            return False
    else:
        return False
