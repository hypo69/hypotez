```python
import requests
import os
from src.logger import logger
from typing import Dict, Any

# URL API
url: str = os.environ.get('API_URL', 'http://127.0.0.1:8443')  # Получение URL из переменной окружения

# Заголовки
headers: Dict[str, str] = {
    'Authorization': f'Bearer {os.environ.get("API_TOKEN", "YOUR_API_TOKEN")}',  # Получение токена из переменной окружения
    'Content-Type': 'application/json'
}

# Данные для отправки
data: Dict[str, str] = {
    'key1': 'value1',
    'key2': 'value2'
}


def send_post_request(url: str, headers: Dict[str, str], data: Dict[str, Any]) -> None:
    """
    Отправляет POST-запрос к указанному URL с заданными заголовками и данными.

    Args:
        url (str): URL для отправки запроса.
        headers (Dict[str, str]): Заголовки запроса.
        data (Dict[str, Any]): Данные для отправки в формате JSON.

    Returns:
        None

    Raises:
        requests.exceptions.RequestException: При возникновении ошибок при отправке запроса.
    """
    try:
        response = requests.post(url, headers=headers, json=data) # Отправка POST-запроса

        if response.status_code == 200: # Проверка статуса ответа
            logger.info(f'Успешно: {response.json()}') # Логирование успешного ответа
        else:
            logger.error(f'Ошибка: {response.status_code}, {response.text}') # Логирование ошибки
    except requests.exceptions.RequestException as ex:
        logger.error(f'Ошибка при отправке запроса: {ex}', exc_info=True) # Логирование исключения


if __name__ == '__main__':
    send_post_request(url, headers, data) # Вызов функции для отправки запроса
