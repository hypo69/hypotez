    - Не закрывается соединение с базой данных после выполнения запросов.
    - Не соблюдены пробелы вокруг операторов присваивания.

**Оптимизированный код**:

```python
import telebot
import sqlite3
import os
from dotenv import load_dotenv
from src.logger import logger  # Добавлен импорт logger
from typing import List

"""
Модуль для отправки промокода пользователям Telegram бота.
=========================================================

Модуль содержит логику для подключения к базе данных, выборки пользователей
и отправки им сообщений с промокодом.
"""


def send_promo_messages():
    """
    Отправляет промо-сообщения пользователям Telegram бота из базы данных.

    Функция подключается к базе данных, выбирает пользователей, у которых не активирован промокод,
    и отправляет им сообщение с информацией о промокоде.
    """
    load_dotenv()
    bot_token: str = os.environ.get('TOKEN')
    if not bot_token:
        logger.error('Не удалось получить токен бота из окружения')
        return

    bot = telebot.TeleBot(token=bot_token)
    conn: sqlite3.Connection | None = None  # Объявляем conn заранее и инициализируем None
    try:
        conn = sqlite3.connect('UsersData.db')
        cursor: sqlite3.Cursor = conn.cursor()

        # Используем execute и fetchall для получения данных
        cursor.execute("SELECT id FROM users_data_table WHERE promocode != 1")
        users: List[tuple[int]] = cursor.fetchall()  # Явное указание типа для users

        for user in users:
            try:
                bot.send_message(
                    chat_id=user[0],
                    text="Успейте воспользоваться промокодом FREE24 до 21 декабря!\n\nПо нему вы получите бесплатный месяц тарифа PRO — это безлимит на генерацию текста и изображений 💥 \n\nЧтобы ввести промокод, перейдите на вкладку Тарифы и нажмите кнопку «Промокод».",
                    parse_mode='html'
                )
                logger.info(f"Сообщение отправлено пользователю {user[0]}")
            except telebot.apihelper.ApiException as ex:
                logger.error(f"Не удалось отправить сообщение пользователю {user[0]}", ex, exc_info=True)
            except Exception as ex:
                logger.error(f"Непредвиденная ошибка при отправке сообщения пользователю {user[0]}", ex, exc_info=True)
            else:
                logger.info(f"Сообщение успешно отправлено пользователю {user[0]}")
    except sqlite3.Error as ex:
        logger.error("Ошибка при работе с базой данных", ex, exc_info=True)
    except Exception as ex:
        logger.error("Непредвиденная ошибка", ex, exc_info=True)
    finally:
        if conn:
            conn.close()  # Закрываем соединение в блоке finally, чтобы оно закрывалось в любом случае

send_promo_messages()
