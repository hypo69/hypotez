**Оптимизированный код:**

```python
"""
Модуль для рассылки промокодов пользователям Telegram.
=======================================================

Модуль содержит функции для подключения к базе данных,
получения списка пользователей и отправки им сообщений
с промокодом через Telegram bot.
"""

import os
import sqlite3
from typing import List, Tuple

import telebot
from dotenv import load_dotenv

from src.logger import logger

load_dotenv()


def send_promocode_to_users() -> None:
    """
    Получает список пользователей из базы данных и отправляет им сообщение с промокодом.

    Args:
        None

    Returns:
        None

    Raises:
        Exception: Если произошла ошибка при подключении к базе данных или отправке сообщения.

    Example:
        >>> send_promocode_to_users()
    """
    token: str = os.environ.get('TOKEN')
    if not token:
        logger.error('TOKEN is not defined in environment variables.')
        return

    bot = telebot.TeleBot(token=token)

    try:
        with sqlite3.connect('UsersData.db') as conn:
            cursor = conn.cursor()
            cursor.execute("SELECT id FROM users_data_table WHERE promocode != 1")
            users: List[Tuple[int]] = cursor.fetchall()

            promocode_text: str = "Успейте воспользоваться промокодом FREE24 до 21 декабря!\n\nПо нему вы получите бесплатный месяц тарифа PRO — это безлимит на генерацию текста и изображений 💥 \n\nЧтобы ввести промокод, перейдите на вкладку Тарифы и нажмите кнопку «Промокод»."

            for user in users:
                try:
                    bot.send_message(chat_id=user[0], text=promocode_text, parse_mode='html')
                    logger.info(f'Message sent to user {user[0]}')
                    print(user[0], "yes")
                except Exception as ex:
                    logger.error(f'Error sending message to user {user[0]}', ex, exc_info=True)
                    print(user[0], "no")
    except sqlite3.Error as ex:
        logger.error('Error connecting to database', ex, exc_info=True)

if __name__ == '__main__':
    send_promocode_to_users()
