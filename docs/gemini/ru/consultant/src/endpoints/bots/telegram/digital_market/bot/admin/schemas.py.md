from pydantic import BaseModel, Field
from typing import Optional
from uuid import UUID


class ProductIDModel(BaseModel):
    """
    Модель для представления ID продукта.

    Args:
        id (int): Уникальный идентификатор продукта.
    """
    id: int


class ProductModel(BaseModel):
    """
    Модель для представления продукта.

    Args:
        name (str): Название продукта (минимум 5 символов).
        description (str): Описание продукта (минимум 5 символов).
        price (int): Цена продукта (должна быть больше 0).
        category_id (int): ID категории продукта (должна быть больше 0).
        file_id (Optional[UUID]): ID файла продукта (может быть None).
        hidden_content (str): Скрытый контент продукта (минимум 5 символов).
    """
    name: str = Field(..., min_length=5, description='Название продукта')
    description: str = Field(..., min_length=5, description='Описание продукта')
    price: int = Field(..., gt=0, description='Цена продукта')
    category_id: int = Field(..., gt=0, description='ID категории продукта')
    file_id: Optional[UUID] = Field(None, description='ID файла продукта')
    hidden_content: str = Field(..., min_length=5, description='Скрытый контент продукта')
