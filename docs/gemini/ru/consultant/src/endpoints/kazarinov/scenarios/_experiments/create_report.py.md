# -*- coding: utf-8 -*-

"""
Модуль для создания отчетов на основе данных, полученных от AI-модели.
==========================================================================

Модуль создает HTML и PDF отчеты на русском и английском языках,
используя данные, предоставленные AI-моделью.
"""

from pathlib import Path
from src.logger import logger # Добавлен импорт logger
import sys

# Проверка наличия test_directory и response_he_dict
try:
    from .ask_model import response_he_dict, test_directory # Явный импорт из .ask_model
except ImportError as ex:

    sys.exit(1) # Завершаем выполнение, если импорт не удался

from src.endpoints.kazarinov.pricelist_generator import ReportGenerator

report_generator = ReportGenerator()

html_file_he: Path = test_directory / 'he.html'
pdf_file_he: Path = test_directory / 'he.pdf'
html_file_ru: Path = test_directory / 'ru.html'
pdf_file_ru: Path = test_directory / 'ru.pdf'

try:
    report_generator.create_report(response_he_dict['he'], 'he', html_file_he, pdf_file_he)
    report_generator.create_report(response_ru_dict['ru'], 'ru', html_file_ru, pdf_file_ru)
except Exception as ex:
    logger.error(f"Ошибка при создании отчета: {ex}", exc_info=True)
