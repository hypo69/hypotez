```python
"""
Модуль для демонстрации и отладки путей в окружении.
=====================================================

Модуль предназначен для вывода значения переменной окружения PATH,
а также обработки возможных ошибок, связанных с отсутствием данной переменной.

Пример использования
----------------------

>>> import os
>>> from src.logger import logger
>>> path = get_path()
>>> if path:
>>>     logger.info(f'Значение переменной PATH: {path}')
"""
import os
from typing import Optional
from src.logger import logger


def get_path() -> Optional[str]:
    """
    Получает значение переменной окружения PATH.

    Returns:
        Optional[str]: Значение переменной PATH или None, если переменная не определена.
    
    Raises:
        KeyError: Если переменная окружения PATH не определена.
    """
    try:
        path: str = os.environ['PATH'] # Получаем значение переменной окружения PATH
        return path
    except KeyError as ex:
        logger.error('Переменная окружения PATH не определена', ex, exc_info=True) # Логируем ошибку, если переменная не определена
        return None


if __name__ == '__main__':
    path = get_path() # Получаем значение переменной PATH
    if path:
        logger.info(f'Значение переменной PATH: {path}') # Выводим значение переменной PATH, если она определена
