"""
Модуль для поиска пути
=================================================

Модуль содержит пример кода для отображения переменной окружения PATH.
"""
import os
from src.logger import logger

def print_path() -> None:
    """
    Выводит значение переменной окружения PATH в лог.

    Args:
        None

    Returns:
        None

    Example:
        >>> print_path()
    """
