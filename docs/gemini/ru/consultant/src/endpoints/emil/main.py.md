## \file /src/endpoints/emil/main.py
# -*- coding: utf-8 -*-
#! .pyenv/bin/python3

"""
Модуль для запуска обработчиков сценариев emil-design.com
=========================================================

Модуль содержит точку входа для запуска основной функции `main` из модуля `emil_design`.
Он обеспечивает запуск приложения при вызове скрипта напрямую.

Пример использования
----------------------

>>> python -m src.endpoints.emil.main
"""

from .emil_design import main as emil_main
from src.logger import logger


def main() -> None:
    """
    Запускает основной процесс обработки сценариев emil-design.com.

    Эта функция является точкой входа для запуска обработки сценариев,
    определенных в модуле `emil_design`. В случае возникновения исключений,
    они логируются.

    Args:
        None

    Returns:
        None

    Raises:
        Exception: Если в процессе выполнения `emil_main()` возникает необработанное исключение.
                   Информация об исключении будет записана в лог.
    """
    try:
        emil_main()
    except Exception as ex:
        logger.error('Ошибка при выполнении main() в модуле emil_design', ex, exc_info=True)


if __name__ == '__main__':
    main()
