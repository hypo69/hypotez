## \file /src/ai/gradio/gradio.py
# -*- coding: utf-8 -*-

#! .pyenv/bin/python3

"""
Модуль для создания простого интерфейса Gradio
==============================================

Этот модуль содержит функцию `greet`, которая принимает имя и возвращает приветствие,
а также создает и запускает интерфейс Gradio для этой функции.
"""

import gradio as gr
from src.logger import logger  # Подключаем модуль для логирования

def greet(name: str) -> str:
    """
    Возвращает приветствие для указанного имени.

    Args:
        name (str): Имя для приветствия.

    Returns:
        str: Приветствие.

    Example:
        >>> greet("World")
        'Hello World!'
    """
    try:
        greeting = "Hello " + name + "!"
        logger.info(f"Создано приветствие для имени: {name}")  # Логируем создание приветствия
        return greeting
    except Exception as ex:
        logger.error(f"Ошибка при создании приветствия для имени: {name}", ex, exc_info=True)  # Логируем ошибку
        return "Ошибка при создании приветствия."  # Возвращаем сообщение об ошибке

if __name__ == '__main__':
    try:
        demo = gr.Interface(fn=greet, inputs="text", outputs="text", title="Приложение для приветствий") # Добавим заголовок
        logger.info("Интерфейс Gradio создан")  # Логируем создание интерфейса
        demo.launch()
        logger.info("Интерфейс Gradio запущен")  # Логируем запуск интерфейса
    except Exception as ex:
        logger.error("Ошибка при запуске интерфейса Gradio", ex, exc_info=True)  # Логируем ошибку при запуске
