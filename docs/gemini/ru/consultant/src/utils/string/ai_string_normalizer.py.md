## Оптимизированный код:

```python
# -*- coding: utf-8 -*-
#! .pyenv/bin/python3

"""
Модуль для нормализации ответов, возвращаемых AI-моделями
=========================================================

Модуль содержит функцию :func:`normalize_answer`, которая используется для удаления нежелательных префиксов и суффиксов
из текстовых ответов, полученных от AI-моделей.

Пример использования
--------------------

>>> from src.utils.string.ai_response_normalizer import normalize_answer
>>> normalized_str = normalize_answer("```html Пример строки <b>с HTML</b> ```")
>>> print(normalized_str)  #  Пример строки <b>с HTML</b>
"""

from typing import List

from src.logger import logger


prefixes: List[str] = [
    '```md\n',
    '```md',
    '```markdown\n',
    '```markdown',
    '```html\n',
    '```html',
    '```\n',
    '```',
]


def normalize_answer(text: str) -> str:
    """
    Удаляет заданные префиксы и суффиксы из строки.

    Args:
        text (str): Строка для нормализации.

    Returns:
        str: Нормализованная строка без префиксов и суффиксов.

    Raises:
        TypeError: Если входной аргумент `text` не является строкой.

    Example:
        >>> normalize_answer("```html Пример строки <b>с HTML</b> ```")
        'Пример строки <b>с HTML</b> '
    """
    if not isinstance(text, str):
        logger.error(f'Входной аргумент должен быть строкой, а не {type(text)}')
        raise TypeError(f'Входной аргумент должен быть строкой, а не {type(text)}')

    # Перебираем префиксы и удаляем их, если они есть в начале строки
    for prefix in prefixes:
        if text.startswith(prefix):
            text = text[len(prefix):]  # Используем срез вместо removeprefix для совместимости с разными версиями Python

            # Если после удаления префикса строка заканчивается на ```, удаляем и это
            if text.endswith('```'):
                text = text[:-3]  # Используем срез вместо removesuffix
            return text

    return text
