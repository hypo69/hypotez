# -*- coding: utf-8 -*-

"""
Модуль для отображения анимированного спиннера в консоли
=========================================================

Модуль содержит функции для отображения спиннера (вращающегося курсора) в консоли,
чтобы имитировать процесс загрузки или ожидания.
"""

import time
import sys
from typing import Generator
from src.logger import logger  # Подключаем модуль логгера


def spinning_cursor() -> Generator[str, None, None]:
    """
    Генератор для циклического перебора символов вращающегося курсора: |, /, -, \\.

    Yields:
        str: Следующий символ в последовательности курсора.

    Example:
        >>> cursor = spinning_cursor()
        >>> next(cursor)
        '|'
        >>> next(cursor)
        '/'
        >>> next(cursor)
        '-'
        >>> next(cursor)
        '\\'
    """
    while True:
        for cursor in '|/-\\':
            yield cursor


def show_spinner(duration: float = 5.0, delay: float = 0.1) -> None:
    """
    Отображает вращающийся курсор в консоли в течение указанного времени.

    Args:
        duration (float): Продолжительность отображения спиннера в секундах. По умолчанию 5.0.
        delay (float): Задержка между сменой символов спиннера в секундах. По умолчанию 0.1.

    Raises:
        Exception: Если возникает ошибка при выводе в консоль.

    Example:
        >>> show_spinner(duration=3.0, delay=0.2)  # Отображает спиннер в течение 3 секунд
    """
    spinner = spinning_cursor()
    end_time = time.time() + duration

    while time.time() < end_time:
        try:
            symbol = next(spinner)
            sys.stdout.write(symbol)  # Выводим следующий символ спиннера
            sys.stdout.flush()  # Принудительно выводим в консоль
            time.sleep(delay)  # Пауза
            sys.stdout.write('\b')  # Возврат на один символ назад для перезаписи
        except Exception as ex:
            logger.error('Ошибка при отображении спиннера', ex, exc_info=True)  # Логируем ошибку


if __name__ == "__main__":
    # Пример использования спиннера в скрипте
    logger.info("Спиннер на 5 секунд:")  # Используем логгер вместо print
    show_spinner(duration=5.0, delay=0.1)
    logger.info("\nГотово!")  # Используем логгер вместо print
