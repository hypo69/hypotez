"""
Модуль для конвертации данных из XLS-файлов в словари
=====================================================

Модуль содержит функцию :func:`xls2dict`, которая преобразует XLS-файл в словарь.

Пример использования
----------------------

>>> from pathlib import Path
>>> file_path = Path('example.xls')
>>> data = xls2dict(file_path)
>>> if data:
...     print(f'Data from XLS: {data}')
"""

from pathlib import Path
from typing import Dict, Optional

from src.utils.xls import read_xls_as_dict # Импортируем функцию для чтения XLS-файлов
from src.logger import logger # Подключаем модуль логгирования

def xls2dict(xls_file: str | Path) -> Optional[Dict]:
    """
    Преобразует XLS-файл в словарь.

    Args:
        xls_file (str | Path): Путь к XLS-файлу.

    Returns:
        Optional[Dict]: Словарь, полученный из XLS-файла, или None в случае ошибки.

    Raises:
        Exception: Если возникает ошибка при чтении файла.

    Example:
        >>> from pathlib import Path
        >>> file_path = Path('example.xls')
        >>> data = xls2dict(file_path)
        >>> if data:
        ...     print(f'Data from XLS: {data}')
    """
    try:
        return read_xls_as_dict(xls_file=xls_file) # Читаем XLS-файл и возвращаем как словарь
    except Exception as ex:
        logger.error('Ошибка при чтении XLS-файла', ex, exc_info=True) # Логируем ошибку
        return None
