"""
Модуль для обработки данных о продуктах с AliExpress.
=======================================================

Модуль содержит функции для парсинга информации о продуктах,
полученных через API AliExpress.
"""

from typing import List, Dict

from src.logger import logger  # Import logger


def parse_product(product: Dict) -> Dict:
    """
    Преобразует данные об одном продукте, извлекая URL маленьких изображений.

    Args:
        product (Dict): Словарь с информацией о продукте.

    Returns:
        Dict: Обновленный словарь с информацией о продукте,
              где `product_small_image_urls` является строкой.
    Raises:
        AttributeError: Если `product.product_small_image_urls` не имеет атрибута `string`.
    """
    try:
        product['product_small_image_urls'] = product['product_small_image_urls']  # Убрал .string, так как это может быть причиной ошибки
        return product
    except AttributeError as ex:
        logger.error('Ошибка при парсинге URL маленьких изображений продукта', ex, exc_info=True)
        return product # или обработать ошибку иначе


def parse_products(products: List[Dict]) -> List[Dict]:
    """
    Преобразует список продуктов, применяя функцию `parse_product` к каждому элементу.

    Args:
        products (List[Dict]): Список словарей с информацией о продуктах.

    Returns:
        List[Dict]: Список обновленных словарей с информацией о продуктах.
    """
    new_products = []

    for product in products:
        new_products.append(parse_product(product))

    return new_products
