from typing import List

import header
from src.suppliers.suppliers_list.aliexpress.scenarios import AliPromoDeal
from src.logger import logger  # Добавлен импорт logger

class PromoteDeal:
    """
    Класс для создания рекламных кампаний на AliExpress.
    """
    def __init__(self, deal_name: str):
        """
        Инициализирует класс PromoteDeal.

        Args:
            deal_name (str): Название рекламной кампании.
        """
        self.deal = AliPromoDeal(deal_name)

    def get_all_products(self) -> List[dict]:
        """
        Получает детали всех продуктов для рекламной кампании.

        Returns:
            List[dict]: Список словарей с деталями продуктов.
        
        Raises:
            Exception: В случае ошибки при получении данных о продуктах.
        """
        try:
            products = self.deal.get_all_products_details()
            logger.info(f"Получено {len(products)} продуктов для рекламной кампании.")  # Логирование
            return products
        except Exception as ex:
            logger.error("Ошибка при получении данных о продуктах", ex, exc_info=True)  # Логирование ошибки
            return []
