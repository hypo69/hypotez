"""
Модуль для определения общих стилей UI элементов
====================================================

Модуль содержит функцию :func:`set_fixed_size`, которая устанавливает фиксированный размер для виджета PyQt6.
"""

from PyQt6 import QtWidgets
from src.logger import logger  # Добавлен импорт logger


def set_fixed_size(widget: QtWidgets.QWidget, width: int, height: int) -> None:
    """
    Устанавливает фиксированный размер для заданного виджета PyQt6.

    Args:
        widget (QtWidgets.QWidget): Виджет, для которого необходимо установить фиксированный размер.
        width (int): Ширина виджета.
        height (int): Высота виджета.

    Returns:
        None

    Example:
        >>> from PyQt6.QtWidgets import QWidget
        >>> widget = QWidget()
        >>> set_fixed_size(widget, 100, 50)
        >>> widget.size().width()
        100
        >>> widget.size().height()
        50
    """
    try:
        widget.setFixedSize(width, height)
    except Exception as ex:
        logger.error('Ошибка при установке фиксированного размера виджета', ex, exc_info=True)
