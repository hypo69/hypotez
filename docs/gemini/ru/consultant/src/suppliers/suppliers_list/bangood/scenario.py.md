def get_list_products_in_category(s) -> Optional[List[str]]:
    """
    Извлекает список URL товаров со страницы категории.

    Args:
        s: Объект Supplier, содержащий информацию о поставщике и драйвер веб-браузера.

    Returns:
        Optional[List[str]]: Список URL товаров на странице категории.
                             Возвращает None, если не удалось получить список товаров.
    """
    driver = s.driver
    locators: dict = s.locators['category']

    # Закрытие всплывающего баннера, если он есть
    driver.execute_locator(s.locators['product']['close_banner'])

    if not locators:
        logger.error('Отсутствуют локаторы для категории')
        return

    driver.scroll()  # Прокрутка страницы для загрузки всех элементов

    # Извлечение ссылок на товары
    product_links = driver.execute_locator(locators['product_links'])

    if not product_links:
        logger.warning('Не найдено ссылок на товары на странице категории')
        return

    # Преобразование в список, если возвращена строка
    product_links = [product_links] if isinstance(product_links, str) else product_links

    logger.info(f'Найдено {len(product_links)} товаров на странице категории')

    return product_links
